Проблема текущего TaskService

Сейчас твой TaskService хорошо подходит для консольного приложения, потому что он возвращает строки вроде:

"Task added: ..."
"Task deleted"
"Task with id ... not found."

Для Web API это уже не очень удобно, потому что API должен возвращать:

HTTP-статусы
JSON-данные
понятные результаты, а не просто строки

Поэтому мы пока сделаем минимально рабочий вариант, не ломая всё сразу.

Шаг 5.1. Добавим метод получения задачи по Id

Сначала в TaskService нужен метод для получения одной задачи.

Открой TaskService.cs и добавь:

public async Task<TaskItem?> GetTaskByIdAsync(int id)
{
    List<TaskItem> tasks = await _repository.GetAllTasksAsync();
    return tasks.FirstOrDefault(t => t.Id == id);
}
Почему пока так

Да, у репозитория уже может быть метод GetTaskByIdAsync, но сейчас нам 
достаточно простого и понятного варианта.

Позже можно будет сделать это аккуратнее через прямой вызов репозитория.

Итоговый TasksController.cs

Вот полный рабочий вариант:

using Microsoft.AspNetCore.Mvc;
using TaskTracker.Api.Models;
using TaskTracker.Api.Services;

namespace TaskTracker.Api.Controllers;

[ApiController]
[Route("api/[controller]")]
public class TasksController : ControllerBase
{
    private readonly TaskService _taskService;

    public TasksController(TaskService taskService)
    {
        _taskService = taskService;
    }

    [HttpGet]
    public async Task<ActionResult<List<TaskItem>>> GetAllTasks()
    {
        List<TaskItem> tasks = await _taskService.GetAllTasksAsync();
        return Ok(tasks);
    }

    [HttpGet("{id:int}")]
    public async Task<ActionResult<TaskItem>> GetTaskById(int id)
    {
        TaskItem? task = await _taskService.GetTaskByIdAsync(id);

        if (task == null)
        {
            return NotFound();
        }

        return Ok(task);
    }

    [HttpPost]
    public async Task<ActionResult> CreateTask([FromBody] CreateTaskRequest request)
    {
        string result = await _taskService.AddTaskAsync(request.Title);

        if (result == "Task title should not be empty.")
        {
            return BadRequest(new { message = result });
        }

        return Ok(new { message = result });
    }

    [HttpPut("{id:int}")]
    public async Task<ActionResult> UpdateTaskTitle(int id, [FromBody] UpdateTaskTitleRequest request)
    {
        string result = await _taskService.UpdateTaskTitleAsync(id, request.Title);

        if (result == "Task title should not be empty.")
        {
            return BadRequest(new { message = result });
        }

        if (result.Contains("not found"))
        {
            return NotFound(new { message = result });
        }

        return Ok(new { message = result });
    }

    [HttpDelete("{id:int}")]
    public async Task<ActionResult> DeleteTask(int id)
    {
        string result = await _taskService.DeleteTaskAsync(id);

        if (result.Contains("not found"))
        {
            return NotFound(new { message = result });
        }

        return Ok(new { message = result });
    }
}
